<script>
let customers = [];
let reminders = [];

function addCustomer(){
  const c = {
    name: document.getElementById("name").value,
    email: document.getElementById("email").value,
    phone: document.getElementById("phone").value,
    amount: document.getElementById("amount").value,
    comment: document.getElementById("comment").value,
    status: document.getElementById("status").value
  };
  customers.push(c);
  renderTables();
  updateReminderDropdown();
}

function renderTables(){
  ["activeTable","inactiveTable","archiveTable","blockedTable"].forEach(id=>{
    document.querySelector("#"+id+" tbody").innerHTML="";
  });
  customers.forEach((c,i)=>{
    let row = `<tr>
      <td>${c.name}</td><td>${c.email}</td><td>${c.phone}</td>
      <td>${c.amount}</td><td>${c.status}</td>
      <td>
        <button class="edit" onclick="editCustomer(${i})">✏️</button>
        <button class="delete" onclick="deleteCustomer(${i})">🗑️</button>
      </td></tr>`;
    if(c.status==="פעיל") document.querySelector("#activeTable tbody").innerHTML+=row;
    else if(c.status==="לא פעיל") document.querySelector("#inactiveTable tbody").innerHTML+=row;
    else if(c.status==="בארכיון") document.querySelector("#archiveTable tbody").innerHTML+=row;
    else if(c.status==="חסום") document.querySelector("#blockedTable tbody").innerHTML+=row;
  });
}
function editCustomer(i){ alert("עריכת לקוח: "+customers[i].name); }
function deleteCustomer(i){ if(confirm("למחוק את "+customers[i].name+"?")){ customers.splice(i,1); renderTables(); } }
function performSearch(){
  let q=document.getElementById("searchInput").value.toLowerCase();
  customers.forEach(c=>{
    if(c.name.toLowerCase().includes(q) || c.email.toLowerCase().includes(q) || c.phone.includes(q)){
      alert("נמצא: "+c.name);
    }
  });
}
function addReminder(){
  const r={
    customer: document.getElementById("reminderCustomer").value,
    date: document.getElementById("reminderDate").value,
    time: document.getElementById("reminderTime").value,
    msg: document.getElementById("reminderMessage").value
  };
  reminders.push(r);
  renderReminders();
}
function renderReminders(){
  document.querySelector("#reminderTable tbody").innerHTML="";
  reminders.forEach((r,i)=>{
    document.querySelector("#reminderTable tbody").innerHTML+=
    `<tr><td>${r.customer}</td><td>${r.date}</td><td>${r.time}</td><td>${r.msg}</td>
    <td><button class="delete" onclick="deleteReminder(${i})">🗑️</button></td></tr>`;
  });
}
function deleteReminder(i){ reminders.splice(i,1); renderReminders(); }
function updateReminderDropdown(){
  let sel=document.getElementById("reminderCustomer");
  sel.innerHTML="";
  customers.forEach(c=>{
    sel.innerHTML+=`<option>${c.name}</option>`;
  });
}
